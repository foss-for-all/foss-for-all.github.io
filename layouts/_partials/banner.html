{{- /* Donation progress banner, rendered site-wide above the navbar. Shows donationbox one-time donations and GitHub Sponsors stacked, plus a combined total as a progress bar. Each source appears only when its data is available; the banner renders nothing when neither is. Amounts are fetched at build time; when HUGO_DONATIONBOX_LIVEURL is set (CORS proxy) they refresh in the browser. The GitHub figure is the current monthly recurring total of PUBLIC sponsors (sponsorshipsAsMaintainer, includePrivate:false), which is monthly while donationbox is cumulative, so the combined figure is approximate. Sponsors USD is converted to KRW at the live ECB rate (api.frankfurter.dev); HUGO_KRW_PER_USD overrides it and 1350 is the fallback. Caps at 100 sponsors (first:100, no pagination). Env: HUGO_DONATIONBOX_UID, HUGO_DONATIONBOX_LIVEURL, HUGO_GITHUB_SPONSORS_ORG, HUGO_GITHUB_TOKEN, HUGO_KRW_PER_USD. */ -}} {{- /* donationbox: cumulative one-time donations */ -}} {{- $db := false -}} {{- $uid := getenv "HUGO_DONATIONBOX_UID" -}} {{- if $uid -}} {{- $api := "https://link.donationbox.co.kr/paysoft_rental-logic" -}} {{- $session := `[{"sessionUserIp":"0.0.0.0","sessionUserId":"","sessionLangType":"ko","sessionAccount":""}]` -}} {{- $opts := dict "method" "post" "headers" (dict "Content-Type" "application/x-www-form-urlencoded") -}} {{- $sumBody := querify "ds_json" (printf `[{"campaignuid":"%s"}]` $uid) "ds_session" $session -}} {{- $res := try (resources.GetRemote (printf "%s/common/getExternCampaignSummary.json" $api) (merge $opts (dict "body" $sumBody))) -}} {{- with $res.Err -}} {{- warnf "donation banner: summary fetch failed: %s" . -}} {{- else with $res.Value -}} {{- $sum := (.Content | transform.Unmarshal).rtnMap -}} {{- $perfBody := querify "ds_json" (printf `[{"campaignkey":%d,"regularflag":"N","irregularflag":"Y"}]` (int $sum.campaignkey)) "ds_session" $session -}} {{- $res = try (resources.GetRemote (printf "%s/common/getCampaignPerformanceSummary.json" $api) (merge $opts (dict "body" $perfBody))) -}} {{- with $res.Err -}} {{- warnf "donation banner: performance fetch failed: %s" . -}} {{- else with $res.Value -}} {{- $perf := (.Content | transform.Unmarshal).rtnMap -}} {{- $db = dict "title" $sum.title "url" (printf "https://link.donationbox.co.kr/donationBoxList.jsp?campaignuid=%s" $uid) "target" (int $sum.irregulartargetamt) "amount" (int $perf.irregulardoneamount) -}} {{- end -}} {{- end -}} {{- end -}} {{- /* GitHub Sponsors: monthly estimate, needs an org-owner token */ -}} {{- $sponsorsEnabled := false -}}{{- /* temporarily disabled; flip to true to re-enable */ -}} {{- $sp := false -}} {{- $token := getenv "HUGO_GITHUB_TOKEN" -}} {{- $org := getenv "HUGO_GITHUB_SPONSORS_ORG" -}} {{- if and $sponsorsEnabled $token $org -}} {{- $query := "query($l:String!){organization(login:$l){sponsorshipsAsMaintainer(first:100,includePrivate:false,activeOnly:true){totalCount nodes{isOneTimePayment tier{monthlyPriceInCents}}}}}" -}} {{- $q := dict "query" $query "variables" (dict "l" $org) -}} {{- $gopts := dict "method" "post" "headers" (dict "Authorization" (printf "Bearer %s" $token) "Content-Type" "application/json") "body" (jsonify $q) -}} {{- $gr := try (resources.GetRemote "https://api.github.com/graphql" $gopts) -}} {{- with $gr.Err -}} {{- warnf "donation banner: github sponsors fetch failed: %s" . -}} {{- else with $gr.Value -}} {{- with ((.Content | transform.Unmarshal).data.organization).sponsorshipsAsMaintainer -}} {{- $cents := 0 -}} {{- range .nodes -}} {{- if not .isOneTimePayment -}}{{- with .tier -}}{{- $cents = add $cents (int .monthlyPriceInCents) -}}{{- end -}}{{- end -}} {{- end -}} {{- $sp = dict "usd" (div (float $cents) 100.0) "count" (int .totalCount) "url" (printf "https://github.com/sponsors/%s" $org) -}} {{- end -}} {{- end -}} {{- end -}} {{- if or $db $sp -}} {{- /* USD->KRW: HUGO_KRW_PER_USD overrides; else live ECB rate; else 1350 */ -}} {{- $krw := 0 -}} {{- with getenv "HUGO_KRW_PER_USD" -}}{{- $krw = int . -}}{{- end -}} {{- if and $sp (eq $krw 0) -}} {{- $fx := try (resources.GetRemote "https://api.frankfurter.dev/v1/latest?base=USD&symbols=KRW") -}} {{- with $fx.Err -}} {{- warnf "donation banner: fx fetch failed, using 1350: %s" . -}} {{- else with $fx.Value -}} {{- with (.Content | transform.Unmarshal).rates.KRW -}}{{- $krw = int (math.Round .) -}}{{- end -}} {{- end -}} {{- end -}} {{- if eq $krw 0 -}}{{- $krw = 1350 -}}{{- end -}} {{- $onetime := 0 -}}{{- with $db -}}{{- $onetime = .amount -}}{{- end -}} {{- $spKrw := 0 -}}{{- with $sp -}}{{- $spKrw = int (math.Round (mul .usd (float $krw))) -}}{{- end -}} {{- $combined := add $onetime $spKrw -}} {{- $target := 0 -}}{{- with $db -}}{{- $target = .target -}}{{- end -}} {{- $cpct := 0 -}}{{- if gt $target 0 -}}{{- $cpct = int (math.Min 100 (math.Round (mul (div (float $combined) (float $target)) 100))) -}}{{- end -}} {{- $donateURL := "" -}}{{- with $db -}}{{- $donateURL = .url -}}{{- else with $sp -}}{{- $donateURL = .url -}}{{- end -}} {{- $aboutURL := "https://forum.fossforall.org/t/foss-for-all-microgrants-2026/105" -}}
{{ T "donation_tagline" }} {{- with $db }} {{ T "donation_onetime" (dict "Amount" (lang.FormatNumberCustom 0 .amount)) | safeHTML }} {{- end }} {{- with $sp }} {{ T "donation_sponsors" (dict "Usd" (lang.FormatNumberCustom 0 .usd) "Count" .count) | safeHTML }} {{- end }} {{- if gt $target 0 }} {{ T "donation_total" (dict "Amount" (lang.FormatNumberCustom 0 $combined) "Target" (lang.FormatNumberCustom 0 $target) "Pct" $cpct) | safeHTML }} {{- end }} {{ T "donation_about" }} {{ T "donation_donate" }}
